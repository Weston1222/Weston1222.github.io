RxJS

全文參考自30 天精通 RxJS

建立 Observable

var observable = Rx.Observable.create(function (observer) {
  observer.next("Jerry");
  observer.next("Anna"); //執行next函式
});

observable.subscribe({
  next: function (value) {
    //定義next函式
    console.log(value);
  },
  error: function (error) {
    console.log(error);
  },
  complete: function () {
    console.log("complete");
  },
});

在訂閱(subscribe)的時候定義next函式，在訂閱的同時會直接執行。

of

of可以用簡潔的方式達成上面create的效果。

var source = Rx.Observable.of("Jerry", "Anna");

source.subscribe({
  next: function (value) {
    console.log(value);
  },
  complete: function () {
    console.log("complete!");
  },
  error: function (error) {
    console.log(error);
  },
});

// Jerry
// Anna
// complete!

其他

takeLast可以取最後幾個，takeLast(2)表示取最後兩個。 last等於takeLast(1)，取最後一個。

combineLatest

combineLatest可以接受多個 observable 和一個 callback function，callback function 的參數量要和放入的 observable 一樣，當 observable 送出值時會被這個 callback function 接收。

每當有一個 observable 送出值就會執行一次 callback function，而其他參數會採用該 observable 最後一次送出的值，如果是剛執行沒有上一個值的話則不執行。

var source = Rx.Observable.interval(500).take(3);
var newest = Rx.Observable.interval(300).take(6);

var example = source.combineLatest(newest, (x, y) => x + y);

example.subscribe({
  next: (value) => {
    console.log(value);
  },
  error: (err) => {
    console.log("Error: " + err);
  },
  complete: () => {
    console.log("complete");
  },
});
// 0
// 1
// 2
// 3
// 4
// 5
// 6
// 7
// complete

zip

zip和combineLatest有點像，只是zip傳入 callback function 的參數是按照順序的，所以當比較短的 Observable 結束時整個就會直接結束，因為不會再有對應的順序。(zip 會有 cache 的問題，建議非必要不要用。)

var source = Rx.Observable.interval(500).take(3);
var newest = Rx.Observable.interval(300).take(6);

var example = source.zip(newest, (x, y) => x + y);

example.subscribe({
  next: (value) => {
    console.log(value);
  },
  error: (err) => {
    console.log("Error: " + err);
  },
  complete: () => {
    console.log("complete");
  },
});
// 0
// 2
// 4
// complete

withLatestFrom

withLatestFrom和combineLatest也有點像，只是要當主要的 observable 送出新的值時 callback function 才會被觸發。

var main = Rx.Observable.from("hello").zip(Rx.Observable.interval(500), (x, y) => x);
var some = Rx.Observable.from([0, 1, 0, 0, 0, 1]).zip(Rx.Observable.interval(300), (x, y) => x);

var example = main.withLatestFrom(some, (x, y) => {
  return y === 1 ? x.toUpperCase() : x;
});

example.subscribe({
  next: (value) => {
    console.log(value);
  },
  error: (err) => {
    console.log("Error: " + err);
  },
  complete: () => {
    console.log("complete");
  },
});

debounceTime

debounceTime每收到一個元素就會等一段時間(下面範例為 1000 毫秒)如果收到新的元素就會丟棄原本的元素;如果在 1000 毫秒內沒有收到新的元素就直接送出。

var source = Rx.Observable.interval(300).take(5);
var example = source.debounceTime(1000);

example.subscribe({
  next: (value) => {
    console.log(value);
  },
  error: (err) => {
    console.log("Error: " + err);
  },
  complete: () => {
    console.log("complete");
  },
});
// 4
// complete

distinct

可以把重複的值濾掉。

var source = Rx.Observable.from(["a", "b", "c", "a", "b"]).zip(Rx.Observable.interval(300), (x, y) => x);
var example = source.distinct();

example.subscribe({
  next: (value) => {
    console.log(value);
  },
  error: (err) => {
    console.log("Error: " + err);
  },
  complete: () => {
    console.log("complete");
  },
});
// a
// b
// c
// complete

也可以傳入一個 callback function 來過濾想要過濾的值。(在 js 中物件的比對比的是記憶體位置)

var source = Rx.Observable.from([{ value: "a" }, { value: "b" }, { value: "c" }, { value: "a" }, { value: "c" }]).zip(Rx.Observable.interval(300), (x, y) => x);
var example = source.distinct((x) => {
  return x.value;
});

example.subscribe({
  next: (value) => {
    console.log(value);
  },
  error: (err) => {
    console.log("Error: " + err);
  },
  complete: () => {
    console.log("complete");
  },
});
// {value: "a"}
// {value: "b"}
// {value: "c"}
// complete
